Backup Procedures
# Create backup script
#!/bin/bash
DATE=$(date +%Y%m%d)
BACKUP_DIR="/home/khoivan/backups"

# Backup website
tar -czf $BACKUP_DIR/web_$DATE.tar.gz /var/www/khoivan.dev

# Backup nginx config
tar -czf $BACKUP_DIR/nginx_$DATE.tar.gz /etc/nginx

# Keep only last 7 days
find $BACKUP_DIR -name "*.tar.gz" -mtime +7 -delete

🐛 TROUBLESHOOTING
Common Issues & Solutions
1. Website không accessible
# Check nginx status
sudo systemctl status nginx

# Restart nginx
sudo systemctl restart nginx

# Check nginx config
sudo nginx -t

# View error logs
sudo tail -f /var/log/nginx/error.log

2. Deploy failed
# Check GitHub Actions log
# Go to repo → Actions → Click failed workflow

# Common fixes:
# - Check SSH key in GitHub Secrets
# - Verify server disk space: df -h
# - Check folder permissions:
sudo chown -R khoivan:khoivan /var/www/khoivan.dev

3. SSH connection refused
# From another terminal or recovery console:
# Check SSH service
sudo systemctl status ssh

# Restart SSH
sudo systemctl restart ssh

# Check port 2222 is open
sudo netstat -tlnp | grep 2222

4. High server load
# Check processes
htop  # or top

# Check disk I/O
iotop

# Check network connections
sudo netstat -tulpn

# Restart services if needed
sudo systemctl restart nginx
